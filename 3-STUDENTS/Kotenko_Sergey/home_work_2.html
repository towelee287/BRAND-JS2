Задание намбер уан(1):

var a = 1, b = 1, c, d;

c = ++a; alert(c); // 2 // - Так как это Префиксная запись. ++ выполняют роль прибавление на 1 так как стоит до a

d = b++; alert(d); // 1 //- Так как это Постфиксная запись. ++ не выполняют роль прибавление на 1 так как стоит после a

c = (2+ ++a); alert(c); // 5 // - Так как ++ идёт до a && a - перезаписалась в прошлом на ++ и получается = 5

d = (2+ b++); alert(d); // 4 // - Так как ++ идёт после b && b - перезаписалась в прошлом на ++ и получается = 4

alert(a); // 3 // - 2 раза перезаписывалась на ++

alert(b); // 3 // - 2 раза перезаписывалась на ++
